LevelLoaded(_)
AND
DB_NightmareMode_StrongerBosses_Initialized(0)
THEN
TimerLaunch("NightmareMode_StrongerBosses_InitDelay", 0);

IF
SavegameLoaded()
AND
DB_NightmareMode_StrongerBosses_Initialized(0)
THEN
TimerLaunch("NightmareMode_StrongerBosses_InitDelay", 0);

IF
TimerFinished("NightmareMode_StrongerBosses_InitDelay")
THEN
NOT DB_NightmareMode_StrongerBosses_Initialized(0);
